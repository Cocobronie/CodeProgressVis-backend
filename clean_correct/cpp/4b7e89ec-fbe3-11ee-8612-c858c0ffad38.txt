cpp class Solution { public: vector<int> searchRange(vector<int>& nums, int target) { int len = nums.size(); if(len == 0) return {-1, -1}; int left = 0, right = len - 1; int begin = -1 , end = -1; while(right >= left) { int mid = left + (right - left) / 2; if(nums[mid] > target) right = mid - 1; else { left = mid + 1; } } if(right >=0 && nums[right] == target) { end = right; while(right >= 1 && nums[right] == nums[right - 1]) right--; begin = right; } return {begin, end}; } };
