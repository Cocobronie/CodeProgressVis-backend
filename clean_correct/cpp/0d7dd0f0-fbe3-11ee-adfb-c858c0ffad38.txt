cpp class Solution { public: vector<int> searchRange(vector<int>& nums, int target) { vector<int> res(2,-1); int len=nums.size(); if(len==0) return res; int left=0,right=len-1; while(left<right) { int mid=(left+right)/2; if(nums[mid]>=target) right=mid; else left=mid+1; } if(nums[left]==target) { res.clear(); res.push_back(left); } else return res; right=len-1; while(left<right) { int mid=(left+right+1)/2; if(nums[mid]<=target) left=mid; else right=mid-1; } res.push_back(right); return res; } };
