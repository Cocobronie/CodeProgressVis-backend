cpp class Solution { public: vector<int> searchRange(vector<int>& nums, int target) { vector<int> d(2,-1); if(nums.size() ==0) {return d;} int l=0, r=nums.size()-1; while(l<r){ int m = l + (r-l)/2; if(nums[m]>=target) r = m; else l = m+1; } vector<int> ans; if(nums[l] ==target){ ans.push_back(l); for(int i = l+1;i<nums.size();i++){ if(nums[i]==nums[l]) ans.push_back(i); } if(ans.size()<=1){ ans.push_back(ans.back()); } if(ans.size()>2){ int a = ans.back(); while(ans.size()>=2){ ans.pop_back(); } ans.push_back(a); } } else {ans.push_back(-1); ans.push_back(-1); } return ans; } };
