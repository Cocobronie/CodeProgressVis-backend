cpp class Solution { public: vector<int> searchRange(vector<int>& nums, int target) { if(nums.size() == 0) return {-1, -1}; int start = binarySearch(nums, target); if(start == nums.size() || nums[start] != target) return {-1, -1}; int end = binarySearch(nums, target + 1); return {start, end - 1}; } int binarySearch(vector<int>& nums, int target){ int left = 0, right = nums.size() - 1; while(left <= right){ int mid = left + (right - left) / 2; if(nums[mid] >= target) right = mid - 1; else left = mid + 1; } return left; } };
