 public int[] searchRange(int[] nums, int target) { int[] ret = {-1, -1}; int len = nums.length; int left = 0; int right = len - 1; int p = -1; while (left <= right) { int mid = (left + right)/2; if (nums[mid] == target) { p = mid; break; } else if (nums[mid] > target) { right = mid - 1; } else { left = mid + 1; } } if (p != -1) { int i = p; int j = p; for (; i>-1; i--) { if (nums[i] != target) break; } for (; j<nums.length; j++) { if (nums[j] != target) break; } ret[0] = i + 1; ret[1] = j - 1; } return ret; } }
