java class Solution { public int[] searchRange(int[] nums, int target) { if(nums.length == 0) return new int[]{-1, -1}; int l = 0, r = nums.length - 1; while(l < r) { int mid = r + l >> 1; if(nums[mid] >= target) r = mid; else l = mid + 1; } int left = l; if(nums[left] != target) return new int[]{-1, -1}; else { l = 0; r = nums.length - 1; while(l < r) { int mid = r + l + 1 >> 1; if(nums[mid] <= target) l = mid; else r = mid - 1; } return new int[]{left, r}; } } }
