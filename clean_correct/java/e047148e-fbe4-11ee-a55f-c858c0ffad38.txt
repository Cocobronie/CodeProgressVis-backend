java class Solution { public int[] searchRange(int[] nums, int target) { int[] ans = new int[2]; if (nums.length == 0)return new int[]{-1, -1}; int l = 0, r = nums.length - 1; while(l < r){ int mid = (l + r) / 2; if (nums[mid] >= target) r = mid; else l = mid + 1; } if (target != nums[r]) ans[0] = -1; else ans[0] = r; l = 0; r = nums.length - 1; while(l < r){ int mid = (l + r + 1) / 2; if (nums[mid] <= target) l = mid; else r = mid - 1; } if (target != nums[r]) ans[1] = -1; else ans[1] = r; return ans; } }
