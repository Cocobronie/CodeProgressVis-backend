java class Solution { public int[] searchRange(int[] nums, int target) { int result[] = {-1,-1}; if(nums.length == 0){ return result; } if(nums.length == 1){ if(nums[0] == target){ return new int[2]; }else{ return result; } } int left = 0; int right = nums.length - 1; while(left <= right){ int mid = (left + right) / 2; if(nums[mid] < target){ left = mid + 1; }else if(nums[mid] > target){ right = mid - 1; }else{ int i = mid - 1; while(i >= 0 && nums[i] == nums[mid]){ i--; } result[0] = i+1; int j = mid + 1; while(j < nums.length && nums[j] == nums[mid]){ j++; } result[1] = j - 1; break; } } return result; } }
