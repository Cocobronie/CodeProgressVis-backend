 int* searchRange(int* nums, int numsSize, int target, int* returnSize){ int *ret = (int *)malloc(sizeof(int)*3); bool isFine = false,lisFine = false; ret[0]=-1;ret[1]=-1; int i=0,j=numsSize-1; *returnSize = 2; while(i<=j){ if(nums[i]==target&&isFine==false){ ret[0] = i; isFine = true; } if(nums[j]==target&&lisFine==false){ ret[1] = j; lisFine = true; } if(nums[i]==target&&nums[j]==target){ i++;j--; continue; } if(isFine==true){ j--; continue; } if(lisFine==true){ i++; continue; } i++; j--; } return ret; }
