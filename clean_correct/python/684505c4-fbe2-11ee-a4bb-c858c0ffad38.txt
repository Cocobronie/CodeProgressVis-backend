python class Solution(object): def searchRange(self, nums, target): """ :type nums: List[int] :type target: int :rtype: List[int] """ i=0 j=len(nums)-1 while i<=j: mid=(i+j)//2 if nums[mid]<target: i=mid+1 elif nums[mid]>target: j=mid-1 else: pre=mid low=mid while pre>=0 and nums[pre]==target: pre-=1 while low<len(nums) and nums[low]==target: low+=1 return [pre+1,low-1] return [-1,-1]
